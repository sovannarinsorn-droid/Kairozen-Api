import telebot, requests, time

bot     = telebot.TeleBot("BOT_TOKEN")
API_KEY = "{{ user.api_key or 'kz_your_key' }}"
BASE    = "https://kairozen-khqr.onrender.com"
HDR     = {"x-api-key": API_KEY}

@bot.message_handler(commands=["pay"])
def pay(msg):
    # 1. Create QR — server auto-polls every 5s
    r = requests.post(f"{BASE}/api/v1/create-qr", headers=HDR, json={
        "amount": 5000, "currency": "KHR",
        "note": "Order #1", "expire_minutes": 10,
        "poll_interval": 5   # server checks every 5 seconds
    })
    data   = r.json()
    pay_id = data["pay_id"]
    qr     = data["data"]["qrString"]
    bot.send_message(msg.chat.id, f"📱 Scan QR:\n`{qr}`", parse_mode="Markdown")

    # 2. Bot-side check loop (optional — server already polling)
    for _ in range(120):   # wait up to 10 min
        s = requests.get(f"{BASE}/api/v1/payment-status/{pay_id}", headers=HDR).json()
        if s["status"] == "PAID":
            bot.send_message(msg.chat.id, "✅ បានបង់ប្រាក់ហើយ!")
            break
        elif s["status"] == "EXPIRED":
            bot.send_message(msg.chat.id, "⏰ QR expired"); break
        time.sleep(5)

bot.polling()
